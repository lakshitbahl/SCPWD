404  private function _uploadFileChunked($path, $writeMode, $inStream, $numBytes, $chunkSize)
405  {
406  Path::checkArg("path", $path);
407  WriteMode::checkArg("writeMode", $writeMode);
408  Checker::argResource("inStream", $inStream);
409  Checker::argNatOrNull("numBytes", $numBytes);
410  Checker::argNat("chunkSize", $chunkSize);
411 
412  // NOTE: This function performs 3 retries on every call. This is maybe not the right
413  // layer to make retry decisions. It's also awkward because none of the other calls
414  // perform retries.
415 
416  assert($chunkSize > 0);
417 
418  $data = self::readFully($inStream, $chunkSize);
419  $len = strlen($data);
420 
421  $client = $this;
422  $uploadId = RequestUtil::runWithRetry(3, function() use ($data, $client) {
423  return $client->chunkedUploadStart($data);
424  });
425 
426  $byteOffset = $len;
427 
428  while (!feof($inStream)) {
429  $data = self::readFully($inStream, $chunkSize);
430  $len = strlen($data);
431 
432  while (true) {
433  $r = RequestUtil::runWithRetry(3,
434  function() use ($client, $uploadId, $byteOffset, $data) {
435  return $client->chunkedUploadContinue($uploadId, $byteOffset, $data);
436  });
437 
438  if ($r === true) { // Chunk got uploaded!
439  $byteOffset += $len;
440  break;
441  }
442  if ($r === false) { // Server didn't recognize our upload ID
443  // This is very unlikely since we're uploading all the chunks in sequence.
444  throw new Exception_BadResponse("Server forgot our uploadId");
445  }
446 
447  // Otherwise, the server is at a different byte offset from us.
448  $serverByteOffset = $r;
449  assert($serverByteOffset !== $byteOffset); // chunkedUploadContinue ensures this.
450  // An earlier byte offset means the server has lost data we sent earlier.
451  if ($serverByteOffset < $byteOffset) throw new Exception_BadResponse(
452  "Server is at an ealier byte offset: us=$byteOffset, server=$serverByteOffset");
453  $diff = $serverByteOffset - $byteOffset;
454  // If the server is past where we think it could possibly be, something went wrong.
455  if ($diff > $len) throw new Exception_BadResponse(
456  "Server is more than a chunk ahead: us=$byteOffset, server=$serverByteOffset");
457  // The normal case is that the server is a bit further along than us because of a
458  // partially-uploaded chunk. Finish it off.
459  $byteOffset += $diff;
460  if ($diff === $len) break; // If the server is at the end, we're done.
461  $data = substr($data, $diff);
462  }
463  }
464 
465  if ($numBytes !== null && $byteOffset !== $numBytes) throw new \InvalidArgumentException(
466  "You passed numBytes=$numBytes but the stream had $byteOffset bytes.");
467 
468  $metadata = RequestUtil::runWithRetry(3,
469  function() use ($client, $uploadId, $path, $writeMode) {
470  return $client->chunkedUploadFinish($uploadId, $path, $writeMode);
471  });
472 
473  return $metadata;
474  }
